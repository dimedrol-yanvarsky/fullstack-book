% =============================================================================
%  preamble.tex — пакеты, настройки, макросы
% =============================================================================

% ---------- Базовый класс ----------
% (загружается в main.tex, здесь только пакеты)

% ---------- Шрифты (fontspec, XeLaTeX) ----------
\usepackage{fontspec}
\setmainfont{CMU Serif}          % основной — кириллица «из коробки»
\setsansfont{CMU Sans Serif}
\setmonofont{JetBrains Mono}[    % моноширинный для листингов
  Scale=0.85,
]
\newfontfamily\cyrillicfonttt{JetBrains Mono}[Scale=0.85]

% ---------- Язык ----------
\usepackage{polyglossia}
\setdefaultlanguage{russian}
\setotherlanguage{english}

% ---------- Математика ----------
\usepackage{amsmath,amssymb,mathtools}

% ---------- Графика ----------
\usepackage{graphicx}
\graphicspath{{images/}}

% ---------- Цвета ----------
\usepackage{xcolor}
\definecolor{codebg}{HTML}{F5F5F5}
\definecolor{codeframe}{HTML}{CCCCCC}
\definecolor{notebg}{HTML}{E8F4FD}
\definecolor{noteframe}{HTML}{2196F3}
\definecolor{importantbg}{HTML}{FFF3E0}
\definecolor{importantframe}{HTML}{FF9800}
\definecolor{linkcolor}{HTML}{1565C0}

% ---------- Листинги кода (minted) ----------
\usepackage{minted}

\setminted{
  style=jsbook,
  fontsize=\small,
  linenos,
  numbersep=8pt,
  frame=single,
  framesep=6pt,
  bgcolor=codebg,
  rulecolor=codeframe,
  breaklines,
  tabsize=2,
}

% Удобное окружение: \begin{code}{javascript} ... \end{code}
\newenvironment{code}[1]
  {\VerbatimEnvironment\begin{minted}{#1}}
  {\end{minted}}

% ---------- Оформление глав ----------
\usepackage{titlesec}
\titleformat{\chapter}[hang]
  {\normalfont\Huge\bfseries}{\chaptertitlename\ \thechapter.\;}{0pt}{}
\titlespacing*{\chapter}{0pt}{-\topskip}{30pt}

% ---------- Оглавление ----------
\setcounter{tocdepth}{3}        % до subsubsection
\setcounter{secnumdepth}{4}     % нумерация до paragraph (1.1.1.1)

\usepackage{tocloft}
\cftsetindents{chap}{0em}{1.5em}        % глава
\cftsetindents{sec}{0.5em}{2.0em}       % раздел
\cftsetindents{subsec}{1.5em}{2.8em}    % подраздел
\cftsetindents{subsubsec}{2.5em}{3.5em} % подподраздел

% ---------- Гиперссылки ----------
\usepackage{hyperref}
\hypersetup{
  colorlinks=true,
  linkcolor=linkcolor,
  urlcolor=linkcolor,
  citecolor=linkcolor,
  bookmarksdepth=3,
  pdfencoding=unicode,
}

% ---------- Геометрия страницы ----------
\usepackage{geometry}
\geometry{headheight=15pt, left=25mm, right=25mm, top=25mm, bottom=25mm}

% ---------- Колонтитулы ----------
\usepackage{fancyhdr}
\pagestyle{fancy}
\fancyhf{}
\fancyhead[L]{\leftmark}        % глава слева
\fancyhead[R]{\rightmark}       % раздел справа
\fancyfoot[C]{\thepage}         % страница по центру
\renewcommand{\headrulewidth}{0.4pt}
\renewcommand{\footrulewidth}{0pt}

% Стиль для первых страниц глав
\fancypagestyle{plain}{
  \fancyhf{}
  \fancyfoot[C]{\thepage}
  \renewcommand{\headrulewidth}{0pt}
}

% ---------- Деревья (forest) ----------
\usepackage{forest}

% ---------- TikZ-библиотеки ----------
\usetikzlibrary{positioning, backgrounds, fit, arrows.meta}

% ---------- Врезки (tcolorbox) ----------
\usepackage[most]{tcolorbox}

\newtcolorbox{notebox}{
  colback=notebg,
  colframe=noteframe,
  fonttitle=\bfseries,
  title={\raisebox{-1pt}{\faInfoCircle}~Примечание},
  boxrule=0.5pt,
  arc=2pt,
  left=6pt, right=6pt, top=4pt, bottom=4pt,
}

\newtcolorbox{importantbox}{
  colback=importantbg,
  colframe=importantframe,
  fonttitle=\bfseries,
  title={\raisebox{-1pt}{\faExclamationTriangle}~Важно},
  boxrule=0.5pt,
  arc=2pt,
  left=6pt, right=6pt, top=4pt, bottom=4pt,
}

% ---------- Иконки ----------
\usepackage{fontawesome5}

% ---------- Кастомные макросы ----------

% \term{русский}{english} → «русский (english)» курсивом
\newcommand{\term}[2]{\textit{#1\/ (\textenglish{#2})}}

% \abbr{ABBR}{Full Name}{пояснение} → «ABBR (Full Name, пояснение)»
\newcommand{\abbr}[3]{\textbf{#1}\/ (\textenglish{#2}, #3)}

% \inl{код} — инлайн-код с серым фоном
\newtcbox{\inl}{
  on line, boxrule=0pt, boxsep=0pt,
  left=3pt, right=3pt, top=2pt, bottom=2pt,
  colback=codebg, colframe=codebg,
  arc=2pt,
  fontupper=\ttfamily\small,
}

% \note{текст} — врезка «Примечание»
\newcommand{\note}[1]{\begin{notebox}#1\end{notebox}}

% \important{текст} — врезка «Важно»
\newcommand{\important}[1]{\begin{importantbox}#1\end{importantbox}}

% ---------- Библиография ----------
\usepackage[
  backend=biber,
  style=numeric,
  sorting=nyt,
]{biblatex}
\addbibresource{references.bib}

% ---------- Прочее ----------
\usepackage{enumitem}           % гибкие списки
\usepackage{booktabs}           % красивые таблицы
\usepackage{microtype}          % микротипографика
